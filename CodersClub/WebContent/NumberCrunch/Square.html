You can break a complex program into several procedures. Also, bear in mind that one procedure can call another.

This program includes a printSquare procedure that uses the smaller and simpler printXs and printSpaces procedures to build up a shape.

Try to modify the program so that, instead of an "x" character, it prints the size of the square to make the square itself. So a square of size 4 would be printed like this:

4444 4 4 4 4 4444

procedure printXs(howMany) { repeat howMany times { print "x"; } } procedure printSpaces(howMany) { repeat howMany times { print " "; } } procedure printSquare(howBig) { call printXs(howBig); new line; number inside = howBig - 2; repeat inside times { print "x"; call printSpaces(inside); print "x"; new line; } call printXs(howBig); new line; } call printSquare(4); new line; call printSquare(7);

function printXs(howMany) { let line = ""; for(let x = 0; x < howMany; ++x) { line = line + "x"; } print(line); } function printSpacedLine(howMany) { let line = "x"; for(let x = 0; x < howMany; ++x) { line = line + " "; } print(line + "x"); } function printSquare(howBig) { printXs(howBig); let inside = howBig - 2; for(let x = 0; x < inside; ++x) { printSpacedLine(inside); } printXs(howBig); } printSquare(4); print(""); printSquare(7);
