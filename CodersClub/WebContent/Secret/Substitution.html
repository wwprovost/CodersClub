Now we'll tackle a new sort of cipher, called a substitution cipher. Instead of a mathetmatical shift in the letter code, now we just map one letter to another: let's say, A becomes G, B becomes X, C becomes M, and so on.

The data for a shift cipher is a single number. For a substitution cipher it's a map, with the original letter as the key and the translated letter as the value. In the example below, notice that H, as it happens, stays H.

message = "HELLO THERE" print message cipher = { "A": "G", "B": "X", "C": "M", "D": "C", "E": "F", "F": "Y", "G": "N", "H": "H", "I": "U", "J": "Q", "K": "B", "L": "A", "M": "K", "N": "V", "O": "W", "P": "R", "Q": "J", "R": "P", "S": "E", "T": "L", "U": "I", "V": "O", "W": "D", "X": "Z", "Y": "S", "Z": "T" } encoded = "" for ltr in message: if ltr in cipher: encoded += cipher[ltr] else: encoded += " " print encoded
let message = "HELLO THERE"; print(message); let cipher = { "A": "G", "B": "X", "C": "M", "D": "C", "E": "F", "F": "Y", "G": "N", "H": "H", "I": "U", "J": "Q", "K": "B", "L": "A", "M": "K", "N": "V", "O": "W", "P": "R", "Q": "J", "R": "P", "S": "E", "T": "L", "U": "I", "V": "O", "W": "D", "X": "Z", "Y": "S", "Z": "T" }; let encoded = ""; for (let ltr of message) { if (ltr in cipher) { encoded += cipher[ltr]; } else { encoded += " "; } } print(encoded);
