A more sophisticated shift cipher rotates the shift value itself as it goes. Consider the program below, in which the initial shift of 1 is increased by 1 for each letter.

So where "ABCDE" would 1-shift to "BCDEF", with this rotating-shift cipher, it's "BDFHJ". Oddly, "EDCBA" would become "FFFFF"!

string message = "ABCDE"; string encoded = ""; number shift = 1; for each letter ltr in message { number cod = code for ltr; if cod != 0 { cod = cod + shift; cod = cod + 25; cod = cod % 26; cod = cod + 1; } string enc = letter for cod; encoded = encoded + enc; shift = shift + 1; } print encoded;
message = "ABCDE" encoded = "" shift = 1 for ltr in message: cod = code(ltr) if cod != 0: cod = cod + shift + 25 cod = cod % 26 + 1 encoded += letter(cod) shift += 1 print encoded
let message = "ABCDE"; let encoded = ""; let shift = 1; for (let ltr of message) { let cod = code(ltr); if (cod != 0) { cod = cod + shift + 25; cod = cod % 26 + 1; } encoded += letter(cod); ++shift; } print(encoded);
